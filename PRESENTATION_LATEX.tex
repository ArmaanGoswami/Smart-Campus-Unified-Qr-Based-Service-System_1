\section{JWT Authentication}
% ============================================================================

\begin{frame}
\frametitle{Why JWT Tokens?}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Session-Based (Old)}
\begin{itemize}
    \item Server stores session data
    \item Doesn't scale well
    \item Mobile unfriendly
    \item Requires server memory
\end{itemize}

\column{0.5\textwidth}
\textbf{JWT Token (New)}
\begin{itemize}
    \item \textcolor{darkgreen}{Stateless}
    \item \textcolor{darkgreen}{Mobile-friendly}
    \item \textcolor{darkgreen}{Instantly verifiable}
    \item \textcolor{darkgreen}{Scales horizontally}
\end{itemize}
\end{columns}

\vspace{1cm}
\textbf{Perfect for Intranet:}
\begin{itemize}
    \item No external identity provider needed
    \item Self-contained security
    \item Offline capability
    \item Token storage in AsyncStorage
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{JWT Token Lifecycle}
\begin{center}
\textbf{1. Login}
$\rightarrow$
\textbf{2. Generate Token}
$\rightarrow$
\textbf{3. Store in Phone}
$\rightarrow$
\textbf{4. Use in Requests}
\end{center}

\vspace{1cm}
\textbf{Token Structure:}
\begin{lstlisting}
Header: { "alg": "HS256", "typ": "JWT" }
Payload: { "sub": "STU123", "role": "STUDENT" }
Signature: HMACSHA256(secret)
\end{lstlisting}

\vspace{0.5cm}
\textbf{Expiry:} 10 hours after generation
\end{frame}

\section{System Architecture}
% ============================================================================

\begin{frame}
\frametitle{Complete Data Flow}
\begin{center}
\small

\textbf{Student Phone}

$\downarrow$ (1. Create Request)

\textbf{Backend API (Port 8080)}

$\downarrow$ (Save to DB)

\textbf{MongoDB}

$\uparrow$ (Notify Warden)

\textbf{Warden Phone}

$\downarrow$ (2. Approve/Reject)

\textbf{Backend API}

$\downarrow$ (Update in DB)

\textbf{MongoDB}

$\uparrow$ (Notify Student)

\textbf{Student Phone} (QR visible)

$\downarrow$ (3. Show to Guard)

\textbf{Guard Phone}

$\downarrow$ (4. Scan QR)

\textbf{Backend API}

$\downarrow$ (Log Access)

\textbf{MongoDB} (Access logged)

\end{center}
\end{frame}

\begin{frame}
\frametitle{API Request Flow with JWT}
\begin{lstlisting}
POST /api/gatepass/request
Authorization: Bearer [token]

Backend Steps:
1. Extract token from Authorization header
2. Verify signature with secret key
3. Check token expiration
4. Extract userId and role from claims
5. Validate user permissions
6. Generate QR code
7. Save to MongoDB
8. Return success response
\end{lstlisting}
\end{frame}

\begin{frame}
\frametitle{Student Flow}
\begin{enumerate}
    \item \textbf{Login:} Enter STU123 / password123
    \item \textbf{Dashboard:} See forms and request history
    \item \textbf{Create Request:} Fill destination, purpose, exit time
    \item \textbf{Submit:} Send to backend
    \item \textbf{Backend:}
    \begin{itemize}
        \item Generate QR code
        \item Create DB record
        \item Notify warden
    \end{itemize}
    \item \textbf{Wait:} Status shows "PENDING"
    \item \textbf{Approved:} QR code visible
    \item \textbf{Gate:} Show QR to guard
\end{enumerate}
\end{frame}

\begin{frame}
\frametitle{Warden Flow}
\begin{enumerate}
    \item \textbf{Login:} Role = WARDEN
    \item \textbf{Pending List:} See all requests needing approval
    \item \textbf{Review:} Click request to see details
    \begin{itemize}
        \item Student name
        \item Destination
        \item Purpose
        \item Requested time
    \end{itemize}
    \item \textbf{Decision:}
    \begin{itemize}
        \item \textcolor{darkgreen}{APPROVE} \textit{Status changes, QR visible to student}
        \item \textcolor{red}{REJECT} \textit{Status changes, reason sent}
    \end{itemize}
    \item \textbf{Log:} Approval recorded in database
\end{enumerate}
\end{frame}

\begin{frame}
\frametitle{Guard Flow}
\begin{enumerate}
    \item \textbf{Login:} Role = GUARD
    \item \textbf{Open Camera:} Guard screen shows live camera
    \item \textbf{Scan QR:} Student shows QR code
    \item \textbf{Verification:}
    \begin{itemize}
        \item Extract pass data from QR
        \item Check if valid
        \item Check if not expired
        \item Check if APPROVED status
    \end{itemize}
    \item \textbf{Result:}
    \begin{itemize}
        \item \textcolor{darkgreen}{\checkmark VERIFIED} \textit{Allow entry}
        \item \textcolor{red}{\(\times\) INVALID} \textit{Deny entry}
    \end{itemize}
    \item \textbf{Log:} Access time recorded
\end{enumerate}
\end{frame}

\section{Development Tools}
% ============================================================================

\begin{frame}
\frametitle{Auto-Sync System}
\textbf{Problem:} Developers had to manually pull code from GitHub every time

\vspace{0.5cm}
\textbf{Solution:} sync.js - Automatic before every start

\begin{lstlisting}
npm start
  -> npm run presync
     -> node sync.js
        - git stash (backup local changes)
        - git pull origin main (get latest)
        - Clear .expo cache
        - Start fresh application
\end{lstlisting}

\vspace{1cm}
\textbf{Benefits:}
\begin{itemize}
    \item Always latest code
    \item No cache issues
    \item Seamless workflow
    \item No manual operations
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{API Fallback Strategy}
\textbf{Multiple Backend URLs with Fallback}

\begin{lstlisting}
const FALLBACK_URLS = [
  "http://render-backend.com",
  "http://192.168.1.100:8080",
  "http://localhost:8080"
];

Try each URL until one responds
\end{lstlisting}

\vspace{1cm}
\textbf{Benefits:}
\begin{itemize}
    \item Works in lab, home, or cloud
    \item Automatic failover
    \item No code changes needed
\end{itemize}
\end{frame}

\begin{itemize}
    \item Backend not running
    \item Check port 8080
    \item Verify CORS config
\end{itemize}

\column{0.5\textwidth}
\textbf{``QR code not scanning''}
\begin{itemize}
    \item Grant camera permission
    \item Proper lighting
    \item Valid QR format
\end{itemize}

\textbf{``Stale code showing''}
\begin{itemize}
    \item Run \texttt{npm start} (auto-syncs)
    \item Clear phone cache
